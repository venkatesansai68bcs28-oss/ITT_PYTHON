#include <iostream>
#include <string>
using namespace std;

int main()
{
    string text, pattern;
    int i = 0, j = 0;
    bool found = false;

    cout << "Enter text: ";

    getline(cin, text);

    cout << "Enter pattern: ";

    getline(cin, pattern);

    int n = text.length();
    int m = pattern.length();


    if (m == 0) {
        cout << "Pattern found successfully at 1 position" << endl;
        return 0;
    }

    while (i < n)
    {
        if (text[i] == pattern[j])
        {
            i++;
            j++;

            if (j == m)
            {
                found = true;
                break;
            }
        }
        else
        {

            i = i - j + 1;
            j = 0;
        }
    }

    if (found)
        cout << "Pattern found successfully at " << i - m + 1 << " position" << endl;
    else
        cout << "Pattern not found" << endl;

    return 0;
}
