#include <iostream>
using namespace std;
int main()
{
   int *arr;
   int n,temp;
   cout << "Enter Number Of Elements:";
   cin >> n;
   arr =new int[n];
   for(int i=0;i<n;i++)
   {
      cin >> arr[i];
   }
   for(int i=0;i<n-1;i++)
   {
      int flag=1;
      for(int j=0;j<n-1-i;j++)
      {
         if(arr[j]>arr[j+1])
         {
            temp=arr[j];
            arr[j]=arr[j+1];
            arr[j+1]=temp;
            flag=0;
         }
      }
      if(flag!=0)
      {
         break;
      }
   }
   cout<<"SORTED ARRAY IS :";
   for(int i=0;i<n;i++)
   {
      cout<<arr[i]<<"\t";
   }
   return 0;
}
